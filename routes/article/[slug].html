Dynamic SSR

Templates and components can be loaded in routes using the @template tag.

  
    <!-- @template article-->

Data can be loaded in routes using the @data tag.

  
    <!-- @data article /article/%slug.json -->
    <simp-article>
      <h2 slot="title">{{ article.title }}</h2>
      <p>{{ article.content }}</p>
    </simp-article>

{{ article.title }}

{{ article.content }}
